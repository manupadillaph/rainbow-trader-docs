from __future__ import annotations

from typing import Any

import polars as pl

from rainbow.common.types.enums import SYNTHETIC_COL


def add_synthetic_stats_to_details(
    details: dict[str, Any],
    frame: pl.DataFrame,
) -> None:
    """Add synthetic count and ratio to validation details.

    Args:
        details: Validation details dict to update (modified in place).
        frame: DataFrame that should have synthetic column.
    """
    if SYNTHETIC_COL not in frame.columns:
        details["stats.synthetic_count"] = 0
        details["stats.synthetic_ratio"] = 0.0
        return

    synthetic_count = int(frame[SYNTHETIC_COL].sum() or 0)
    synthetic_ratio = frame[SYNTHETIC_COL].mean() or 0.0

    details["stats.synthetic_count"] = synthetic_count
    details["stats.synthetic_ratio"] = round(float(synthetic_ratio), 6)
