    def _deep_merge(base: dict[str, object], override: dict[str, object]) -> dict[str, object]:
    """Recursively merge override into base. Override wins for non-dict values.
    For nested dicts, recurse. Lists are REPLACED (not appended)."""
    merged = deepcopy(base)
    for key, value in override.items():
        if key in merged and isinstance(merged[key], dict) and isinstance(value, dict):
            merged[key] = _deep_merge(cast(dict[str, object], merged[key]), cast(dict[str, object], value))
        else:
            merged[key] = deepcopy(value)
    return merged

    Task 5.1: Rewrite _build_optuna_config() (L1184-1329)

    Current: Constructs the entire optuna_config dict by reading flat dataclass fields + hardcoded values. Separate branches for trainable vs non-trainable.

    New: Returns the pre-merged optuna_config dict from TierDefinition, with only these runtime adjustments: 1. Inject combined_confidence_weights_config (from root config) 2.     def _build_optuna_config(
    *,
    spec: ExperimentSpec,
    tier: TierDefinition,
    config: Script2ConfigurationDefinition,
    feature_bounds: FeatureBounds | None = None,
) -> dict[str, object]:
    """Build category-aware Optuna config from pre-merged tier config."""

    # Select pre-merged config by category
    if spec.model_category == "trainable":
        base = deepcopy(tier.trainable_optuna_config)
    elif spec.model_category == "rule_baseline":
        base = deepcopy(tier.rule_baseline_optuna_config)
    else:
        base = deepcopy(tier.random_baselines_optuna_config)

    hp = base["hyperparameters_config"]  # already a dict from deep_merge

    # 1. Override n_trials from spec (spec.n_trials is authoritative)
    base["n_trials"] = spec.n_trials

    # 2. Inject model_space (single model per experiment)
    hp["model_space"] = {"type": "categorical", "choices": [spec.model_type]}

    # 3. Filter model_hyperparameters_config to selected model only
    all_model_hp = hp["model_hyperparameters_config"]
    hp["model_hyperparameters_config"] = {spec.model_type: all_model_hp[spec.model_type]}

    # 4. For random models: inject random_seed_base into each model's hyperparams
    if spec.model_category == "random_baselines":
        model_hp = hp["model_hyperparameters_config"][spec.model_type]
        model_hp["random_seed_base"] = tier.random_seed_base

    # 5. Inject combined_confidence_weights_config
    hp["combined_confidence_weights_config"] = {
        "probability_weight": {"type": "fixed", "value": config.combined_confidence_weights.probability_weight},
        "strength_weight": {"type": "fixed", "value": config.combined_confidence_weights.strength_weight},
        "quality_weight": {"type": "fixed", "value": config.combined_confidence_weights.quality_weight},
        "alpha": {"type": "fixed", "value": config.combined_confidence_weights.alpha},
    }

    # 6. Resolve feature bounds (auto or from preflight)
    fs = hp["feature_selection_config"]
    if feature_bounds is not None:
        fs["min_features"] = feature_bounds[0]
        fs["max_features"] = feature_bounds[1]
    else:
        configured_min = fs.get("min_features")
        configured_max = fs.get("max_features")
        if configured_min == "auto":
            fs["min_features"] = 1  # placeholder, preflight resolves later
        if configured_max == "auto":
            fs["max_features"] = None

    return base
