    Graph Orchestrator - Findings & Specifications

    Fecha: 2025-12-02
    Status: DOCUMENTO FINAL - Listo para implementación
    Scope: modules/graph-orchestrator/, modules/common/, modules/database/

    1. ARQUITECTURA CORE

    1.1 Modelo Conceptual

    ┌─────────────────────────────────────────────────────────────────┐
│                         EVENTS (DB)                              │
│  Persistentes, representan hechos históricos                     │
│  Flujo: REQUESTED → STARTED → PROGRESS* → DONE | FAILED         │
└─────────────────────────────────────────────────────────────────┘
                              ↓ trigger
┌─────────────────────────────────────────────────────────────────┐
│                     ACTIVE OPERATIONS (DB)                       │
│  Transitorias, representan tasks en progreso                      │
│  Status: pending → running → completed/failed/timeout            │
└─────────────────────────────────────────────────────────────────┘
                              ↓ run on
┌─────────────────────────────────────────────────────────────────┐
│                        THREADS (LangGraph)                       │
│  Contexto de ejecución con checkpointer                          │
│  Formato: rainbow_main | exp-{id} | task-{id} | nested...        │
└─────────────────────────────────────────────────────────────────┘

    1.2 Thread ID - Árbol Jerárquico

                                  rainbow_main (root)
                                    │
           ┌────────────────────────┼────────────────────────┐
           │                        │                        │
       exp-1                   task-research            task-refactor
           │                        │                        │
    ┌──────┴──────┐          ┌──────┴──────┐          ┌──────┴──────┐
    │             │          │             │          │             │
 trial-1      trial-2     analyze      summarize    audit       implement
    │             │          │                        │             │
    │        ┌────┴────┐     │                   ┌────┴────┐        │
    │        │         │     │                   │         │        │
backtest  train    validate deep-dive        module1  module2    test

    Formato de thread_id:

    # Patrones válidos
"exp-1"                                    # Experiment
"exp-1.trial-5"                            # Trial
"exp-1.trial-5.training"                   # Trial subtask
"exp-1.trial-5.backtest.analysis"          # Nested subtask
"task-research-abc"                        # Generic task
"task-research-abc.data-collection"        # Task subtask
"task-refactor.audit.module-database"      # Deep nesting

# Construcción
def build_thread_id(*parts: str) -> str:
    return ".".join(parts)

    Relación Thread ↔ Operations:

    Thread 1:N Operations

"exp-1.trial-5" (thread)
   ├── op-uuid-111 (training)
   ├── op-uuid-222 (backtest)
   └── op-uuid-333 (analysis)

    1.3 Async Tool Flow (interrupt/resume)

    ┌─────────────────────────────────────────────────────────────────┐
│                     GRAPH (Agent Node)                           │
│  1. Agent calls request_training()                               │
│  2. Tool emits TOOL_TRAINING_REQUESTED                           │
│  3. Tool creates ActiveOperation in DB                           │
│  4. Tool calls interrupt() → Graph PAUSES                        │
└─────────────────────────────────────────────────────────────────┘
                              │
                              │ State saved to checkpointer
                              │ Process can terminate
                              ▼
┌─────────────────────────────────────────────────────────────────┐
│                    EXTERNAL WORKER                               │
│  1. Receives REQUESTED event                                     │  
│  2. Emits TOOL_TRAINING_STARTED                                  │
│  3. Does actual work (or mock with sleep)                        │

    # graph_orchestrator/event_bridge.py

class EventBridge:
    """Polls events and resumes paused graphs."""

    async def run(self):
        """Main entry point - runs both loops in parallel."""
        await asyncio.gather(
            self._event_poll_loop(),
            self._status_check_loop(),
        )

    async def _event_poll_loop(self):
        """Poll unprocessed events and resume graphs."""
        while not self._stopped:
            events = await self._get_unprocessed_events()

            for event in events:
                thread_id = self._resolve_thread_id(event)
                if thread_id:
                    await self._resume_graph(thread_id, event)
                    await self._mark_processed(event.id)

            await asyncio.sleep(self.poll_interval)  # 1s

    async def _status_check_loop(self):
        """Monitor active operations for timeouts."""
        while not self._stopped:
            operations = await self.operation_service.get_pending_timeout()

            for op in operations:
                await self._handle_timeout(op)

            await asyncio.sleep(self.status_check_interval)  # 30s

    async def _resume_graph(self, thread_id: str, event: Event):
        """Resume a paused graph with event result."""
        result = event.payload

        self.event_service.emit_event(
            EventType.GRAPH_THREAD_RESUMED,
            {"thread_id": thread_id, "resume_event_id": event.id}
        )

        # CASCADE RESPUESTA:
        # ¡Exacto! Los eventos *_DONE son los que disparan el resume.
        # El flow es:
        #
        # 1. Tool llama interrupt() → graph pausa, state guardado
        # 2. Worker externo (o mock) emite *_DONE event → inserted en DB
        # 3. Event Bridge poll loop detecta evento unprocessed
        # 4. Event Bridge llama _resume_graph() con el payload del *_DONE
        # 5. graph.ainvoke(Command(resume=payload)) → graph despierta
        # 6. El interrupt() original RETORNA el payload
        # 7. Agent continúa con el resultado
        #
        # Solo eventos relevantes disparan resume:
        # - TOOL_*_DONE, TOOL_*_FAILED (async tools)
        # - HITL_APPROVAL_DECIDED (human approval)
        # - Otros eventos solo se loggean, no resumen graphs

        await self.graph.ainvoke(
            Command(resume=result),
            config={"configurable": {"thread_id": thread_id}},
        )

    7. HITL PATTERN

    class HITLGuard:
    """Human-in-the-loop using interrupt/resume."""

    def request_approval(
        self,
        action: str,
        context: dict,
    ) -> dict:
        """Request HITL approval and wait for decision."""

        if not self.requires_hitl(action):
            return {"approved": True, "auto": True}

        # Emit request event
        event_id = self.event_service.emit_event(
            EventType.HITL_APPROVAL_REQUESTED,
            {
                "action": action,
                "description": self._get_description(action),
                "context": context,
                "allowed_decisions": ["approve", "edit", "reject"],
            }
        )

        # Interrupt and wait
        decision = interrupt({
            "hitl_action": action,
            "event_id": event_id,
            "context": context,
        })

        # Emit decision event
        self.event_service.emit_event(
            EventType.HITL_APPROVAL_DECIDED,
            {"action": action, "decision": decision}
        )

        return decision
