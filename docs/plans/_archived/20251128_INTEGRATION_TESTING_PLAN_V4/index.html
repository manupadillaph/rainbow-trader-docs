def run_backtest(experiment_id: int, trial_id: int) -> dict:
    """Ejecuta backtest para un trial."""

    # 1. Cargar configuración
    with get_session() as database:
        trial = db.query(Trial).filter_by(trial_id=trial_id).first()
        experiment = db.query(Experiment).filter_by(experiment_id=experiment_id).first()

    # 2. Cargar modelo entrenado
    model = load_checkpoint(trial.checkpoint_path)

    # 3. Cargar bot
    bot = BotEngine(trial.bot_config)

    # 4. Cargar datos históricos
    ohlcv = load_ohlcv(experiment.symbol, experiment.timeframe, experiment.time_range)

    # 5. Estado de simulación
    positions = []
    trades = []
    equity = trial.initial_capital

    # 6. Loop por cada timestamp
    for i, timestamp in enumerate(ohlcv.timestamps):
        # Preparar ventana
        window = prepare_window(ohlcv, i, trial.features, trial.window_size)

        # Predecir
        prediction = model.predict(window)  # {"signal": "LONG", "confidence": 0.85}

        # Bot decide — recibe TODO lo que necesita
        orders = bot.decide(
            prediction=prediction,
            positions=positions,
            current_price=ohlcv.close[i],
            equity=equity,
            market_data={
                "volatility": calc_volatility(ohlcv, i),
                "atr": calc_atr(ohlcv, i),
            }
        )

        # Simular ejecución
        for order in orders:
            trade = simulate_order(order, ohlcv, i)
            trades.append(trade)
            positions = update_positions(positions, trade)
            equity = update_equity(equity, trade)

    # 7. Guardar resultados
    with get_session() as database:
        for trade in trades:
            db.add(BacktestTrade(trial_id=trial_id, **trade))
        db.add(BacktestResult(trial_id=trial_id, final_equity=equity, sharpe=calc_sharpe(trades)))
        db.commit()

    # 8. Emitir evento
    emit_event("BACKTEST_DONE_TRIAL", {"experiment_id": experiment_id, "trial_id": trial_id})

    return {"trial_id": trial_id, "final_equity": equity, "trades": len(trades)}

    def __init__(self, execution_plan_id: int):
        self.execution_plan_id = execution_plan_id
        self.running = False
        self.paused = False

    async def start(self):
        """Inicia trading en vivo."""

        # 1. Cargar plan de ejecución
        with get_session() as database:
            plan = db.query(ExecutionPlan).filter_by(id=self.execution_plan_id).first()

        # 2. Cargar modelo
        self.model = load_checkpoint(plan.checkpoint_path)

        # 3. Cargar bot
        self.bot = BotEngine(plan.bot_config)

        # 4. CARGAR HISTORIAL AL INICIO — no esperar a que se llenen velas
        historical = await fetch_historical(
            symbol=plan.symbol,
            timeframe=plan.timeframe,
            limit=plan.window_size + 100
        )
        self.window_buffer = historical[-plan.window_size:]

        # 5. Estado
        self.positions = []
        self.running = True

        # 6. Conectar a stream de datos (data-fetcher provee async generator)
        async for candle in stream_candles(plan.symbol, plan.timeframe):
            if not self.running:
                break
            if self.paused:
                await asyncio.sleep(1)
                continue

            # Actualizar buffer (ya tiene ventana completa desde inicio)
            self.window_buffer.append(candle)
            self.window_buffer.pop(0)

            # Preparar ventana
            window = prepare_live_window(self.window_buffer, plan.features)

            # Predecir
            prediction = self.model.predict(window)

            # Bot decide — recibe TODO lo que necesita
            orders = self.bot.decide(
                prediction=prediction,
                positions=self.positions,
                current_price=candle.close,
                equity=self.get_equity(),
                market_data={
                    "bid": get_bid(),
                    "ask": get_ask(),
                    "spread": get_spread(),
                    "volatility": calc_volatility(self.window_buffer),
                    "atr": calc_atr(self.window_buffer),
                }
            )

            # Ejecutar orders (REAL o SIMULATED según plan.mode)
            for order in orders:
                if plan.mode == "LIVE_REAL":
                    trade = await execute_real_order(order, plan.exchange_config)

    from dataclasses import dataclass
from typing import List, Optional

@dataclass
class Order:
    action: str  # "OPEN" | "CLOSE"
    side: Optional[str] = None  # "LONG" | "SHORT"
    size: Optional[float] = None
    entry: Optional[float] = None
    stop_loss: Optional[float] = None
    take_profit: Optional[float] = None
    position_id: Optional[int] = None
    reason: Optional[str] = None

class BotEngine:
    def __init__(self, bot_config: dict):
        self.config = bot_config
        self.min_confidence = bot_config.get("min_confidence", 0.7)
        self.max_positions = bot_config.get("max_positions", 3)
        self.sl_atr_mult = bot_config.get("sl_atr_multiplier", 2.0)
        self.tp_atr_mult = bot_config.get("tp_atr_multiplier", 3.0)
        self.risk_per_trade = bot_config.get("risk_per_trade", 0.02)

    def decide(
        self,
        prediction: dict,
        positions: list,
        current_price: float,
        equity: float,
        market_data: dict
    ) -> List[Order]:
        """
        Decide qué órdenes ejecutar basado en predicción y estado actual.

        Args:
            prediction: {"signal": "LONG"|"SHORT"|"NONE", "confidence": 0.0-1.0}
            positions: Lista de posiciones abiertas
            current_price: Precio actual del activo
            equity: Capital disponible
            market_data: {"atr": float, "volatility": float, "bid": float, "ask": float}

        Returns:
            Lista de órdenes a ejecutar
        """
        orders = []
        atr = market_data.get("atr", current_price * 0.01)

        # 1. ¿Debo CERRAR posiciones existentes?
        for pos in positions:
            # ¿Llegó a SL?
            if pos.side == "LONG" and current_price <= pos.stop_loss:
                orders.append(Order(action="CLOSE", position_id=pos.id, reason="SL"))
            elif pos.side == "SHORT" and current_price >= pos.stop_loss:
                orders.append(Order(action="CLOSE", position_id=pos.id, reason="SL"))
            # ¿Llegó a TP?
            elif pos.side == "LONG" and current_price >= pos.take_profit:
                orders.append(Order(action="CLOSE", position_id=pos.id, reason="TP"))
            elif pos.side == "SHORT" and current_price <= pos.take_profit:
                orders.append(Order(action="CLOSE", position_id=pos.id, reason="TP"))
            # ¿Señal opuesta?
            elif prediction["signal"] == "SHORT" and pos.side == "LONG":
                orders.append(Order(action="CLOSE", position_id=pos.id, reason="SIGNAL_FLIP"))
            elif prediction["signal"] == "LONG" and pos.side == "SHORT":
                orders.append(Order(action="CLOSE", position_id=pos.id, reason="SIGNAL_FLIP"))

        # 2. ¿Debo ABRIR posición nueva?
        if prediction["signal"] in ["LONG", "SHORT"]:
            if prediction["confidence"] >= self.min_confidence:
                if len(positions) < self.max_positions:
                    # Calcular tamaño basado en equity y riesgo
                    size = self._calc_position_size(equity, atr, current_price)

                    # Calcular SL/TP basado en ATR
                    if prediction["signal"] == "LONG":
                        sl = current_price - (atr * self.sl_atr_mult)
                        tp = current_price + (atr * self.tp_atr_mult)
                    else:  # SHORT
                        sl = current_price + (atr * self.sl_atr_mult)
                        tp = current_price - (atr * self.tp_atr_mult)

                    orders.append(Order(
                        action="OPEN",
                        side=prediction["signal"],
                        size=size,
                        entry=current_price,
                        stop_loss=sl,
                        take_profit=tp,
                    ))

        return orders

    def _calc_position_size(self, equity: float, atr: float, price: float) -> float:
        """Calcula tamaño de posición basado en riesgo."""
        risk_amount = equity * self.risk_per_trade
        risk_per_unit = atr * self.sl_atr_mult
        size = risk_amount / risk_per_unit
        return round(size, 4)
