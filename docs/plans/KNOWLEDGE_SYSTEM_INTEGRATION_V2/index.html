    Finding 9: Token Tracking Bug (CRITICAL)

    Problem: Budget is ALWAYS $0.00 because tokens are never tracked after LLM calls.

    Current Flow (BROKEN):

    1. _get_llm() → ensure_within_budget(current=0.0) → Always OK
2. llm.invoke() → Tokens consumed (REAL)
3. ❌ NOTHING → Tokens NOT recorded
4. Next call → budget still $0.00

    Decision: Implement in BaseAgent (centralized):

    def _invoke_llm_with_tracking(self, messages, state):
    response = self._llm.invoke(messages)

    # Extract tokens
    token_usage = TokenUsage.from_response(response)

    # Update budget_stats
    new_budget = update_budget_stats(
        state["budget_stats"],
        tokens=token_usage.total_tokens,
        cost_usd=token_usage.cost_usd,
        tier=self.config.tier,
        provider=self.tier_config.provider,
    )

    # Persist to DB
    BudgetGuard.update_after_llm_call(new_budget)

    return response, new_budget

    Location: graph-orchestrator/agents/base.py

    Finding 10: Pre-Call Token Estimation

    Decision: Add estimate_before_call() to BudgetGuard:

    def estimate_before_call(self, prompt: str) -> EstimatedTokens:
    """Estimate tokens and cost BEFORE LLM call using tiktoken."""
    enc = tiktoken.get_encoding("cl100k_base")
    prompt_tokens = len(enc.encode(prompt))
    estimated_output = int(prompt_tokens * 0.5)
    total = prompt_tokens + estimated_output
    cost = (total / 1000) * self.cost_per_1k
    return EstimatedTokens(tokens=total, cost_usd=cost)

    Usage in invoke flow:

    1. estimate_before_call(prompt) → estimated tokens/cost
    2. ensure_within_budget(estimated_cost) → check if budget allows
    3. If budget warning → auto-downgrade tier
    4. If budget exceeded → force offline or HITL

    Location: graph-orchestrator/helpers/budget.py

    For complete flow, see INVOKE_FLOW_ARCHITECTURE.md Section 1.

    Finding 11: Conflict Resolution Flow

    Decision: AUTOMATIC resolution. HITL only for budget extension.

    Scenario Result
    No internet FORCE offline mode

    Finding 13: RAG Híbrido — Embeddings + BM25

    Decision: Combine semantic search + keyword search:

    def hybrid_search(query: str, top_k: int = 5) -> list[Document]:
    # 1. Semantic search (embeddings via Ollama)
    semantic = vector_store.similarity_search(query, k=top_k * 2)

    # 2. Keyword search (BM25 via SQLite FTS5)
    keyword = bm25_index.search(query, k=top_k * 2)

    Finding 15: Decision Cache

    Decision: Automatic cache based on embedding similarity.

    • Threshold: > 95% similarity
    • TTL automatic: GET queries = 5 min, Analysis = 1 hour
    • No cache: CREATE/UPDATE operations
    class DecisionCache:
    def get_cached(self, query_embedding: list[float]) -> str | None:
        for cached in self.cache.values():
            if cosine_similarity(query_embedding, cached.embedding) > 0.95:
                if not cached.is_expired():
                    return cached.decision
        return None

    Location: graph-utils/cache/decision_cache.py

    def hybrid_search(self, query: str, collection: str | None = None, top_k: int = 5) -> list[dict]:
        """Combine semantic search (embeddings) + keyword search (BM25)."""
        # 1. Semantic search
        query_embedding = self.embeddings.embed_text(query)
        semantic_results = self.vector_store.similarity_search(
            collection=collection or "docs",
            query_embedding=query_embedding,
            k=top_k * 2
        )

        # 2. Keyword search (BM25)
        keyword_results = self.bm25.search(query, collection=collection, k=top_k * 2)

        # 3. Reciprocal Rank Fusion
        return self._reciprocal_rank_fusion(semantic_results, keyword_results, k=top_k)

    def _reciprocal_rank_fusion(self, list1: list, list2: list, k: int = 5, rrf_k: int = 60) -> list:
        """Combine two ranked lists using RRF."""
        scores = {}

        for rank, doc in enumerate(list1):
            doc_id = doc.get("id")
            scores[doc_id] = scores.get(doc_id, 0) + 1 / (rrf_k + rank + 1)

        for rank, doc in enumerate(list2):
            doc_id = doc.get("id")
            scores[doc_id] = scores.get(doc_id, 0) + 1 / (rrf_k + rank + 1)

        # Sort by combined score
        all_docs = {d["id"]: d for d in list1 + list2}
        sorted_ids = sorted(scores.keys(), key=lambda x: scores[x], reverse=True)[:k]

        return [all_docs[doc_id] for doc_id in sorted_ids if doc_id in all_docs]

            "content": content,

from datetime import datetime, timedelta
import numpy as np

@dataclass
class CachedDecision:
    embedding: list[float]
    decision: str
    timestamp: datetime
    ttl_seconds: int

    def is_expired(self) -> bool:
        return datetime.now() > self.timestamp + timedelta(seconds=self.ttl_seconds)

class DecisionCache:
    """Cache decisions based on embedding similarity."""

    def __init__(self, similarity_threshold: float = 0.95):
        self.cache: dict[str, CachedDecision] = {}
        self.threshold = similarity_threshold

# ✅ NOTA: DecisionCache es un helper puro en memoria, no necesita config_obj.
# El threshold puede venir de config pero se pasa como parámetro.
# Quien lo instancie debe obtener el threshold de config:
#
# config_obj = Config.load_config()
# threshold = config_obj.get("graph_orchestrator", {}).get("cache", {}).get("similarity_threshold", 0.95)
# cache = DecisionCache(similarity_threshold=threshold)

    def get_cached(self, query_embedding: list[float]) -> str | None:
        """Find similar decision in cache."""
        for key, cached in list(self.cache.items()):
            if cached.is_expired():
                del self.cache[key]
                continue

            similarity = self._cosine_similarity(query_embedding, cached.embedding)
            if similarity > self.threshold:
                return cached.decision

        return None

    def cache_decision(self, query: str, embedding: list[float], decision: str, ttl_seconds: int = 3600) -> None:
        """Cache a decision."""
        self.cache[query] = CachedDecision(
            embedding=embedding,
            decision=decision,
            timestamp=datetime.now(),
            ttl_seconds=ttl_seconds,
        )

    def get_ttl_for_query(self, query: str) -> int:
        """Determine TTL based on query type."""
        query_lower = query.lower()

        # No cache for mutations
        if any(word in query_lower for word in ["create", "new", "update", "delete", "add"]):
            return 0  # Don't cache

        # Short cache for status queries
        if any(word in query_lower for word in ["status", "estado", "list", "show"]):
            return 300  # 5 minutes

        # Long cache for analysis
        if any(word in query_lower for word in ["analyze", "analizar", "explain", "why"]):
            return 3600  # 1 hour

        return 1800  # 30 minutes default

    @staticmethod
    def _cosine_similarity(a: list[float], b: list[float]) -> float:
        a_arr = np.array(a)
        b_arr = np.array(b)
        return float(np.dot(a_arr, b_arr) / (np.linalg.norm(a_arr) * np.linalg.norm(b_arr)))
